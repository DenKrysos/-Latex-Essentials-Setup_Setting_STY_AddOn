% (Silbentrennung)
% 	Usage: \texttthyph{}

        {\not@math@alphabet\ttfamily\mathtt
         \fontfamily\ttdefault\selectfont\hyphenchar\font=-1\relax}
\makeatother
\DeclareTextFontCommand{\texttthyph}{\ttfamily\hyphenchar\font=45\relax}
