\def\DenKrGlsPrintAKA#1{%
    \ifglshasfield{longAlt}{#1}{%
        \normalfont%
        \textit{\DenKrRefLabelAKA}: \glsentrylongAlt{#1}%
        \ifglshasfield{textAlt}{#1}{%
            \ (\glsentrytextAlt{#1})%
        }{}%
        .\space%
    }{}%
}%

\newcommand*{\glsprintmitdeutsch}{%
	{\glsentrylong{\glslabel} (\glsentryname{\glslabel}, dt.:
	\glsentrytextDE{gls-\glslabel})}%
}%
\newcommand*{\glsprintmitenglisch}{%
	{\glsentrylong{\glslabel} (\glsentryname{\glslabel}, engl.:
	\glsentrytextEN{gls-\glslabel})}%
}%
\newcommand*{\glsprintmitdualentry}{%
  \ifglsused{\glslabel}%
    {%
    	\glsgenentryfmt%
    }{%
    	\ifglshasfield{textDE}{gls-\glslabel}{%
    		\glsprintmitdeutsch%
    	}{%
	    	\ifglshasfield{textEN}{gls-\glslabel}{%
	    		\glsprintmitenglisch%
	    	}{%
	    		{\glsgenentryfmt}%
	    	}%
    	}%
    }%
}%
\newcommand*{\glsprintohnedeutsch}{%
  \ifglsused{\glslabel}%
    {\glsgenentryfmt}% Wenn verwendet normales format
    {\glsgenentryfmt}% beim ersten mal anders formatiert
}%
% Alternativen für erste Formatierung:
% textsf, textit, texttt
%
\defglsentryfmt[\acronymtype]{%
	\ifglsentryexists{gls-\glslabel}{%
		\glsprintmitdualentry%
	}{%
		\glsprintohnedeutsch%
	}%
}%

\newglossarystyle{list+crossRef+alt+otherLang+subtitle+url}{%
	\renewenvironment{theglossary}%
		{\begin{myglossarylist}}{\end{myglossarylist}}%
	\renewcommand*{\glossaryheader}{}%
	\renewcommand*{\glsgroupheading}[1]{}%
	\renewcommand*{\glossentry}[2]{%
		\item[\glsentryitem{##1}%
			  \glstarget{##1}{\glossentryname{##1}}]%
%
		\ifglshasfield{subtitle}{##1}{\normalfont%
			\textbf{(}\glsentrysubtitle{##1}\textbf{)}.\space}{}%
		\DenKrGlsPrintAKA{##1}%
		\DenKrGlsPrintOtherLang{##1}%
		\glossentrydesc{##1}\glspostdescription%
        \space##2%
        \ifglshasfield{crossRef}{##1}{%
            .\space%
            \textit{\DenKrRefLabelAcronyms}: \glstext{\glsentrycrossRef{##1}}%
        }{}%
        \ifglshasfield{url}{##1}{\glspar
    		\url{\glsentryurl{##1}}}{}%
    	}%
	\renewcommand*{\subglossentry}[3]{%
		\begin{myglossarylist}%
			\item[\glssubentryitem{##2}%
			\glstarget{##2}{\strut\glossentryname{##2}%
			}]%
%
			\ifglshasfield{subtitle}{##2}{\normalfont%
				\textbf{(}\glsentrysubtitle{##2}\textbf{)}.\space}{}%
		      \DenKrGlsPrintAKA{##2}%
		      \DenKrGlsPrintOtherLang{##2}%
			\glossentrydesc{##2}\glspostdescription%
            \space##3%
            \ifglshasfield{crossRef}{##1}{%
                .\space%
                \textit{\DenKrRefLabelAcronyms}: \glstext{\glsentrycrossRef{##1}}%
            }{.}%
		\end{myglossarylist}%
	}%
	\renewcommand*{\glsgroupskip}{\ifglsnogroupskip\else\indexspace\fi}%
}%
